# --- 1. Definición de la empresa y descarga de datos ---
ticker_empresa <- "ANTO.L"
ticker_mercado <- "^GSPC" # S&P 500 como proxy del mercado

# Descargar últimos 5 años con frecuencia diaria, semanal y mensual
# Usamos tq_get de tidyquant, que se conecta a Yahoo Finance
data_diaria_r <- tq_get(c(ticker_empresa, ticker_mercado), from = Sys.Date() - years(5), to = Sys.Date(), periodicity = "daily")
data_semanal_r <- tq_get(c(ticker_empresa, ticker_mercado), from = Sys.Date() - years(5), to = Sys.Date(), periodicity = "weekly")
data_mensual_r <- tq_get(c(ticker_empresa, ticker_mercado), from = Sys.Date() - years(5), to = Sys.Date(), periodicity = "monthly")

print("Datos descargados para ANTO.L y ^GSPC en frecuencias diaria, semanal y mensual.")

# --- 2. Función para calcular Beta y estadísticas ---
calculate_beta_r <- function(asset_data, market_data, frequency_name, ticker_empresa, ticker_mercado) {
  # Asegurarse de que tenemos los precios de cierre y renombrar columnas
  asset_prices <- asset_data %>% filter(symbol == ticker_empresa) %>% select(date, adjusted)
  market_prices <- market_data %>% filter(symbol == ticker_mercado) %>% select(date, adjusted)

  # Unir los precios por fecha
  combined_prices <- inner_join(asset_prices, market_prices, by = "date", suffix = c(".asset", ".market"))

  # Calcular retornos logarítmicos
  returns <- combined_prices %>%
    mutate(
      Return.asset = log(adjusted.asset / lag(adjusted.asset)),
      Return.market = log(adjusted.market / lag(adjusted.market))
    ) %>% na.omit()

  if (nrow(returns) < 2) {
    message(paste("No hay suficientes datos para calcular beta para", frequency_name))
    return(list(
      "Rentabilidad Promedio" = NA_real_,
      "Riesgo (Desviación Estándar)" = NA_real_,
      "Beta" = NA_real_
    ))
  }

  # Regresión lineal para calcular Beta
  model <- lm(Return.asset ~ Return.market, data = returns)
  beta <- coef(model)[["Return.market"]]

  # Calcular rentabilidad promedio y riesgo
  rentabilidad_promedio <- mean(returns$Return.asset, na.rm = TRUE)
  riesgo_std <- sd(returns$Return.asset, na.rm = TRUE)

  return(list(
    "Rentabilidad Promedio" = rentabilidad_promedio,
    "Riesgo (Desviación Estándar)" = riesgo_std,
    "Beta" = beta
  ))
}

# Calcular beta para cada frecuencia
results_r <- list()

print("\n--- Calculando Beta y Estadísticas ---")
results_r[['Diaria']] <- calculate_beta_r(data_diaria_r, data_diaria_r, "Diaria", ticker_empresa, ticker_mercado)
results_r[['Semanal']] <- calculate_beta_r(data_semanal_r, data_semanal_r, "Semanal", ticker_empresa, ticker_mercado)
results_r[['Mensual']] <- calculate_beta_r(data_mensual_r, data_mensual_r, "Mensual", ticker_empresa, ticker_mercado)

# Convertir resultados a un DataFrame para una mejor visualización
df_results_r <- do.call(rbind, lapply(results_r, as.data.frame))
row.names(df_results_r) <- names(results_r)

print("\n--- Resumen de Rentabilidad, Riesgo y Beta ---")
print(df_results_r)

# Guardar df_results_r para el reporte RMarkdown
saveRDS(df_results_r, "df_results_r.rds")
